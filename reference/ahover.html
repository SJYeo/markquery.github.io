.ahover

link, hover

기본 링크 스타일 text-decoration: underline; 대신에 예쁜 링크 스타일을 만든다. 마우스오버 시 border가 표시된다. 개별 링크에만 스타일이 적용된다. 콘텐츠 전체에 마우스오버 링크 스타일을 적용할 떄는 .nestedhover를 사용. 

// LessCSS
.ahover(@LinkColor, @HoverColor,@BorderColor) {
     color:@LinkColor;
     text-decoration: none;
     padding-bottom: 2px;
  &:hover,
  &:focus {
          color: @HoverColor;
          text-decoration: none;
          border-bottom: 1px solid @BorderColor;
          padding-bottom: 2px;
  }
}

a.ahover-08c-005580 { 
  .ahover(#08c, #005580, #0d65a1);
}
